// Cole este código no Apps Script (script.google.com)
// Planilha: crie em sheets.new e adicione aba "DB_Lotes"

const SHEET_NAME = 'DB_Lotes';

function doGet(e) {
  const acao = e.parameter.acao;
  if (acao === 'ping') return resp({ok:true, msg:'Cacaueiro online'});
  if (acao === 'lerLotes') {
    const sh = getSheet();
    const rows = sh.getDataRange().getValues();
    const lotes = [];
    for (let i = 1; i < rows.length; i++) {
      if (rows[i][0]) {
        try { lotes.push(JSON.parse(rows[i][1])); } catch(e) {}
      }
    }
    return resp({ok:true, lotes});
  }
  return resp({ok:false, msg:'Acao invalida'});
}

function doPost(e) {
  const data = JSON.parse(e.postData.contents);
  const sh = getSheet();
  if (data.acao === 'upsertLote') {
    const lote = data.lote;
    const rows = sh.getDataRange().getValues();
    let found = -1;
    for (let i = 1; i < rows.length; i++) {
      if (rows[i][0] === lote.id) { found = i + 1; break; }
    }
    const now = new Date().toISOString();
    if (found > 0) {
      sh.getRange(found, 1, 1, 3).setValues([[lote.id, JSON.stringify(lote), now]]);
    } else {
      sh.appendRow([lote.id, JSON.stringify(lote), now]);
    }
    return resp({ok:true});
  }
  if (data.acao === 'excluirLote') {
    const rows = sh.getDataRange().getValues();
    for (let i = 1; i < rows.length; i++) {
      if (rows[i][0] === data.id) { sh.deleteRow(i + 1); break; }
    }
    return resp({ok:true});
  }
  return resp({ok:false});
}

function getSheet() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sh = ss.getSheetByName(SHEET_NAME);
  if (!sh) {
    sh = ss.insertSheet(SHEET_NAME);
    sh.appendRow(['ID','DadosJSON','AtualizadoEm']);
  }
  return sh;
}

function resp(data) {
  return ContentService.createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}
